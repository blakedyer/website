import pandas as pd

__all__ = ["load_data"]


[docs]def load_data(filename):
    """
    Import and process data for model creation. Refer to
    :ref:`data formatting <data>`
    for more information.

    Parameters
    ----------
    filename: string or path
        Location of data csv file.

    Returns
    -------
    data: dict
        Dictionary of data for model creation organized by regions.

    """

    data_pd = pd.read_csv(filename)
    data_pd.columns = data_pd.columns.str.lower()
    region_names = data_pd['inference_region'].unique()
    data = {}
    for region in region_names:
        data_rg = data_pd[data_pd['inference_region'] == region]
        data[region] = {}
        data[region]['includes_holocene'] = 'holocene' in list(
            data_rg['interglacial'].values)
        data[region]['includes_lig'] = 'lig' in list(
            data_rg['interglacial'].values)
        data[region]['age'] = data_rg['age'].values
        data[region]['age_1sigma'] = data_rg['age_1sigma'].values
        data[region]['elevation'] = data_rg['elevation'].values
        data[region]['elevation_1sigma'] = data_rg['elevation_1sigma'].values
        data[region]['os_age'] = data_rg['os_age'].values
        data[region]['os_age_1sigma'] = data_rg['os_age_1sigma'].values
        data[region]['lat'] = data_rg['lat'].values
        data[region]['lon'] = data_rg['lon'].values
        data[region]['type'] = data_rg['type'].values
        data[region]['ir_distribution'] = data_rg['ir_distribution'].values
        data[region]['param_1_name'] = data_rg['param_1_name'].values
        data[region]['param_2_name'] = data_rg['param_2_name'].values
        data[region]['param_1'] = data_rg['param_1'].values
        data[region]['param_2'] = data_rg['param_2'].values
        data[region]['emulator_name'] = data_rg['emulator_name'].values
        data[region]['updown_rate'] = data_rg['updown_rate'].values
        data[region]['updown_rate_1sigma'] = data_rg['updown_rate_1sigma'].values
        data[region]['custom_age_distribution'] = data_rg['custom_age_distribution'].values
        data[region]['age_dist_offset'] = data_rg['age_dist_offset'].values
        data[region]['age_dist_scale'] = data_rg['age_dist_scale'].values
        data[region]['age_param_1_name'] = data_rg['age_param_1_name'].values
        data[region]['age_param_1'] = data_rg['age_param_1'].values
        data[region]['age_param_2_name'] = data_rg['age_param_2_name'].values
        data[region]['age_param_2'] = data_rg['age_param_2'].values
        data[region]['sample id'] = data_rg['sample id'].values


    return data
