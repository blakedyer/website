[docs]def observations(data,full_trace):
    """
    A plot showing the predicted elevation prior (unfilled), posterior (filled), and observed (red; :math:`\mu\pm2\sigma`) elevations of each sample.

    .. plot::

        from lig_sea_level.config import PROJECT_ROOT
        from lig_sea_level.inference import load_data
        from lig_sea_level.plotting import observations

        import arviz as az
        full_trace = az.from_netcdf(PROJECT_ROOT/'examples/example.nc')
        data = load_data(PROJECT_ROOT/'data/example.csv')

        observations(data,full_trace)
        plt.show()




    Parameters
    ----------
    data: dict
        A dictionary from :ref:`inference.load_data`.
    full_trace: arviz.InferenceData
        An  :class:`arviz.InferenceData` object containing the full set of prior, prior predictive, posterior, and posterior predictive samples from :py:meth:`get_model() <lig_sea_level.model>` in :py:mod:`lig_sea_level.model`.



    """
    fig = plt.figure(figsize=(12*2, 9*2))
    total_N = 0
    for region in data.keys():
        total_N+=data[region]['age'].size
    sq_side = np.ceil(total_N**(1/2)).astype(int)

    region='Bahamas'

    i = 1
    holN=0
    ligN=0
    for region in data.keys():
        holN+=(data[region]['age']<15).size
        ligN+=(data[region]['age']<15).size
    pal = {}
    pal['hol'] = sns.color_palette('flare',n_colors=holN)
    pal['lig'] = sns.color_palette('crest',n_colors=ligN)

    for sample_name in data[region]["sample id"]:


        plt.subplot(sq_side, sq_side, i)
        i += 1
        param = "elevation_prediction"
        plot_args = {}
        plot_args["prior"] = {"warn_singular": False, "gridsize": 200}
        plot_args["posterior"] = {
            "warn_singular": False,
            "shade": True,
            "alpha": 0.5,
            "gridsize": 200,
            "lw": 3,
        }
        plot_args["posterior_predictive"] = plot_args["posterior"]

        idx = np.where(data[region]["sample id"] == sample_name)[0][0]
        observed_elevation = data[region]["elevation"][idx]
        observed_e_1sig = data[region]["elevation_1sigma"][idx]

        k = "prior"
        region = "Bahamas"
        if data[region]["age"][idx] > 15:
            ig = "lig"
        else:
            ig = "hol"

        ig_thresh = {"lig": operator.gt, "hol": operator.lt}
        ig_index = np.where(ig_thresh[ig](data[region]["age"], 15))[0]

        # ig_index = np.where(data['Bahamas']['age']>15)[0]
        ig_index = np.where(data[region]["sample id"][ig_index] == sample_name)[0][0]
        sample_gia = full_trace[k][f"{region}_{ig}_{param}"].values[:, :, ig_index].ravel()
        sns.kdeplot(sample_gia, **plot_args['prior'], color=pal[ig][ig_index])

        k = "posterior_predictive"
        region = "Bahamas"
        if data[region]["age"][idx] > 15:
            ig = "lig"
        else:
            ig = "hol"

        ig_thresh = {"lig": operator.gt, "hol": operator.lt}
        ig_index = np.where(ig_thresh[ig](data[region]["age"], 15))[0]

        ig_index = np.where(data[region]["sample id"][ig_index] == sample_name)[0][0]
        sample_gia = full_trace[k][f"{region}_{ig}_{param}"].values[:, :, ig_index].ravel()
        sns.kdeplot(sample_gia, **plot_args["posterior"], color=pal[ig][ig_index])

        if (
            np.std(
                full_trace["prior"][f"{region}_{ig}_{param}"].values[:, :, ig_index].ravel()
            )
            > 10
        ):
            plt.gca().set_xlim(
                [
                    np.min(
                        np.floor(
                            full_trace["posterior_predictive"][
                                f"{region}_{ig}_{param}"
                            ].values[:, :, ig_index]
                        )
                    ),
                    np.max(
                        np.ceil(
                            full_trace["posterior_predictive"][
                                f"{region}_{ig}_{param}"
                            ].values[:, :, ig_index]
                        )
                    ),
                ]
            )

        plt.gca().set_title(sample_name)
        plt.gca().set_yticks([])
        plt.gca().set_ylabel(None)
        plt.gca().set_xlabel(None)

        plt.gca().twinx()
        plt.plot(2 * [observed_elevation], [0, 1], "r-",label='Observed\nelevation')
        plt.plot(2 * [observed_elevation-2*observed_e_1sig], [0, 1], "r--")
        plt.plot(2 * [observed_elevation+2*observed_e_1sig], [0, 1], "r--")
        plt.gca().set_ylim([0, 1])
        plt.gca().set_yticks([])
        if i == 2:
            plt.legend(loc='best',frameon=False, fontsize=9)
    fig.tight_layout()

            if (

                        np.min(
                            np.floor(

                                    v
                                ].values[:, :, idx]
                            )

                            np.ceil(

[docs]def get_id(data,sample_name,region):
    """
    Helper function to get the prior and posterior trace index that correctly matches the given ``sample id``.
    """
    idx = np.where(data[region]['sample id'] == sample_name)[0][0]
    if data[region]['age'][idx] > 15:
        ig = 'lig'
    else:
        ig = 'hol'
    ig_index = np.where(data[region]['age']>15)[0]
    ig_index = np.where(data[region]['sample id'] == sample_name)[0][0]
    return ig_index, ig
